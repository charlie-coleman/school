	\chaptertitle{RS 232}
	
	\ti{Signal Levels: }
		
		\hi\ti{Standard: }
			
			\hi\hi High Voltage (Space/False): 3 $\leftrightarrow$ 20V
			
			\hi\hi Low Voltage (Mark/True): -3 $\leftrightarrow$ -20V
			
		\hi\ti{TTL: } 
		
			\hi\hi High Voltage (True): 3.3V or 5V
		
			\hi\hi Low Voltage (False): 0V
	
	\ti{Waveforms: }
		
		\hi Start bit $\rightarrow$ a single 0
		
		\hi Data bits $\rightarrow$ 5-9 bits, LSB first
		
		\hi Parity bit $\rightarrow$ 0-1 bits, even/odd/none/etc.
		
		\hi Stop bits $\rightarrow$ 1 or more bits, can be fractional
	
	\ti{Handshaking: }
	
		\hi\ti{Hardware: } 
		
		\hi RTS \ra output from DTE, indicates ready to communicate
		
		\hi CTS \ra input to the DTE, indicates DCE is ready to communicate
		
		\hi\ti{Software: } XON \ra 0x11, XOFF \ra 0x13
		
	\ti{UART: }
